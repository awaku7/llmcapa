#!/usr/bin/env python3
"""
Scrape LM Studio model catalog from https://lmstudio.ai/models

Usage: python _scrape_lmstudio.py [--save]
"""

import json
import os
import re
import sys
from playwright.sync_api import sync_playwright

OUTPUT = os.path.join(os.path.dirname(__file__), "..", "src", "llmcapa", "data", "lmstudio.json")
BASE_URL = "https://lmstudio.ai"


def extract_model_data(page, slug):
    """Extract all model data from a single LM Studio model detail page."""
    page.goto(f"{BASE_URL}{slug}", wait_until="networkidle", timeout=30000)
    page.wait_for_timeout(2000)

    main = page.query_selector("main") or page.query_selector("body")
    lines = main.inner_text().split("\n")
    lines = [l.strip() for l in lines if l.strip()]

    # Find key sections
    sections = {}
    section_headers = ["Models", "Capabilities", "About", "Memory Requirements", "License"]
    for h in section_headers:
        try:
            idx = lines.index(h)
            content = []
            for i in range(idx + 1, len(lines)):
                if lines[i] in section_headers:
                    break
                if lines[i] in ("Product", "Download the app"):
                    break
                content.append(lines[i])
            sections[h] = content
        except ValueError:
            sections[h] = []

    # -- Extract name --
    # Name is typically the line before "Downloads" or a standalone line
    name = ""
    for i, line in enumerate(lines):
        if line == slug.split("/")[-1].replace("-", " ").title():
            name = line
            break
    if not name:
        # Try to find the line right before "Downloads"
        for i, line in enumerate(lines):
            if "Downloads" in line and i > 0:
                name = lines[i - 1]
                break

    # -- Extract download count --
    downloads = ""
    for line in lines:
        m = re.search(r"([\d.]+[KMB]?)\s*Downloads", line)
        if m:
            downloads = m.group(1)
            break

    # -- Extract model variants --
    # Variants appear after "Models" heading: model_id, size, download_count, "Get"
    variants = []
    if "Models" in sections and sections["Models"]:
        model_lines = sections["Models"]
        i = 0
        while i < len(model_lines):
            line = model_lines[i]
            # A variant model_id looks like org/model-name
            if "/" in line and not line.startswith("To run"):
                model_id = line
                size = ""
                dl_count = ""
                if i + 1 < len(model_lines) and "GB" in model_lines[i + 1]:
                    size = model_lines[i + 1]
                    if i + 2 < len(model_lines) and model_lines[i + 2].isdigit():
                        dl_count = model_lines[i + 2]
                variants.append({
                    "model_id": model_id,
                    "size_gb": size,
                    "downloads": dl_count,
                })
            i += 1

    # -- Extract capabilities description --
    capabilities_raw = " ".join(sections.get("Capabilities", []))
    about_raw = " ".join(sections.get("About", []))

    # Determine capabilities
    supports_vision = any(w in capabilities_raw.lower() or w in about_raw.lower()
                          for w in ["vision", "image", "multimodal", "video"])
    supports_reasoning = any(w in capabilities_raw.lower() or w in about_raw.lower()
                             for w in ["reasoning", "thinking"])
    supports_function_calling = any(w in capabilities_raw.lower() or w in about_raw.lower()
                                     for w in ["tool use", "function calling", "tools"])
    input_modalities = ["text"]
    output_modalities = ["text"]
    if supports_vision:
        input_modalities.append("image")
        if "video" in capabilities_raw.lower() or "video" in about_raw.lower():
            input_modalities.append("video")
    if "audio" in capabilities_raw.lower() or "audio" in about_raw.lower():
        input_modalities.append("audio")

    # -- Extract context window --
    context_window = 0
    full_text = capabilities_raw + " " + about_raw
    # Look for patterns like "128K tokens", "context window of up to 256K"
    ctx_matches = re.findall(r'(\d+)\s*[Kk]\s*[Tt]oken', full_text)
    if ctx_matches:
        context_window = max(int(m) * 1024 for m in ctx_matches)
    if not context_window:
        ctx_matches = re.findall(r'context\s*(?:window|length)?\s*(?:of\s*)?(?:up\s*to\s*)?(\d+)\s*[Kk]', full_text)
        if ctx_matches:
            context_window = max(int(m) * 1024 for m in ctx_matches)

    # -- Extract max output tokens --
    max_output = 0
    out_matches = re.findall(r'(\d+)\s*[Kk]\s*output', full_text)
    if out_matches:
        max_output = max(int(m) * 1024 for m in out_matches)

    # -- Build entries for each variant --
    display_name_base = name
    result_variants = []
    seen_ids = set()
    for v in variants:
        mid = v["model_id"]
        if not mid or mid in seen_ids:
            continue
        seen_ids.add(mid)

        # Create a display name from the variant
        # Extract the last part of model_id for display
        parts = mid.split("/")
        variant_name = parts[-1] if len(parts) > 1 else mid
        # Remove org prefix from display
        dname = variant_name.replace("-", " ").replace("_", " ").title()
        # If the base name appears in the display name, use just the size info
        base_words = display_name_base.lower().replace("-", " ").split()
        # Simplified: just use variant_name as display
        display = variant_name

        entry = {
            "provider": "lmstudio",
            "model_id": mid,
            "display_name": display,
            "context_window": context_window,
            "max_output_tokens": max_output if max_output else 8192,
            "input_modalities": input_modalities,
            "output_modalities": output_modalities,
            "supports_function_calling": supports_function_calling,
            "supports_json_mode": True,
            "supports_streaming": True,
            "supports_vision": supports_vision,
            "supports_reasoning": supports_reasoning,
            "knowledge_cutoff": "2024-01-01",
            "pricing": {
                "input_per_1m": 0.0,
                "output_per_1m": 0.0,
                "currency": "USD"
            },
            "aliases": [],
            "deprecated": False,
            "supports_fim": False,
            "license_type": "free",
        }
        result_variants.append(entry)

    return result_variants


def main():
    save = "--save" in sys.argv

    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True)
        page = browser.new_page()

        # Get all model family links
        print("Fetching model list from main page...")
        page.goto(f"{BASE_URL}/models", wait_until="networkidle", timeout=30000)
        page.wait_for_timeout(2000)
        slugs = page.evaluate("""() => {
            const links = document.querySelectorAll('a');
            const modelLinks = new Set();
            for (const link of links) {
                const href = link.getAttribute('href');
                if (href && href.startsWith('/models/') && href !== '/models') {
                    modelLinks.add(href);
                }
            }
            return Array.from(modelLinks);
        }""")
        print(f"Found {len(slugs)} model families: {slugs}")
        page.close()

        # Scrape each model family page
        all_models = []
        seen_model_ids = set()
        for idx, slug in enumerate(slugs):
            print(f"\n[{idx + 1}/{len(slugs)}] Scraping {slug}...")
            try:
                spage = browser.new_page()
                variants = extract_model_data(spage, slug)
                spage.close()
                print(f"  -> {len(variants)} variant(s)")
                for v in variants:
                    mid = v["model_id"]
                    if mid not in seen_model_ids:
                        seen_model_ids.add(mid)
                        all_models.append(v)
                    else:
                        print(f"  SKIP (duplicate): {mid}")
            except Exception as e:
                print(f"  ERROR: {e}")

        browser.close()

    all_models.sort(key=lambda m: m["model_id"])
    print(f"\nTotal unique model entries: {len(all_models)}")

    output_data = {"models": all_models}

    if save:
        os.makedirs(os.path.dirname(OUTPUT), exist_ok=True)
        with open(OUTPUT, "w", encoding="utf-8") as f:
            json.dump(output_data, f, indent=2, ensure_ascii=False)
        print(f"Saved to {OUTPUT}")
    else:
        print(f"\nPreview (first 5):")
        for m in all_models[:5]:
            print(f"  {m['model_id']:50s} | ctx={m['context_window']} | vision={m['supports_vision']} | tools={m['supports_function_calling']} | reasoning={m['supports_reasoning']}")
        print(f"\nRun with --save to write to {OUTPUT}")


if __name__ == "__main__":
    main()
